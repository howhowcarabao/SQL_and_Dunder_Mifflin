-- Find the total sales of each salesman

FROM
  `my-data-project-11097972.Dundler_Miffin.works_with`

-- Find the total amount of money spent by each client
SELECT
  DISTINCT (client_id),
  SUM(total_sales) AS total_purchase

  `my-data-project-11097972.Dundler_Miffin.works_with`

---- JOINS
-- Find the client name and total amount of money spent by each client
SELECT
  DISTINCT (`my-data-project-11097972.Dundler_Miffin.works_with`.client_id),
  client_name,
  SUM(`my-data-project-11097972.Dundler_Miffin.works_with`.total_sales) AS total_purchase

FROM
  `my-data-project-11097972.Dundler_Miffin.works_with`
JOIN
  `my-data-project-11097972.Dundler_Miffin.client` -- LEFT JOIN
ON 
`my-data-project-11097972.Dundler_Miffin.works_with`.client_id = `my-data-project-11097972.Dundler_Miffin.client`.client_id
GROUP BY
client_id,
client_name
ORDER BY 
total_purchase DESC

-- Find the first name of the total sales of each salesman
SELECT
  DISTINCT (works_with.emp_id) AS sales_man_id,
  first_name,
  SUM(works_with.total_sales) AS total_sales
FROM
  `my-data-project-11097972.Dundler_Miffin.works_with` AS works_with --(ALIAS)
JOIN                                                                             
    `my-data-project-11097972.Dundler_Miffin.employee` AS employee   --(ALIAS)  
ON 
works_with.emp_id = employee.emp_id
GROUP BY
sales_man_id,
first_name
ORDER BY 
total_sales DESC

---- Nested Queries
-- Find names of all employees who have sold over 50,000
SELECT
 first_name,        -- Second step: build the outer query
 last_name
FROM 
`my-data-project-11097972.Dundler_Miffin.employee`
WHERE emp_id IN (

                  SELECT              -- first step: build the inner query
                      emp_id
                  FROM
                      `my-data-project-11097972.Dundler_Miffin.works_with`
                  WHERE
                        total_sales >= 50000)

-- Find all clients who are handles by the branch that Michael Scott manages
(Assume you know Michael's ID - 102)
SELECT
client_name
FROM
`my-data-project-11097972.Dundler_Miffin.client`    -- build outer query
WHERE
branch_id IN (
               SELECT
                 branch_id
               FROM
                `my-data-project-11097972.Dundler_Miffin.branch`  -- build inner query
                WHERE
                 mgr_id = 102)

-- Find all clients who are handles by the branch that Michael Scott manages
(Assume you DONT'T know Michael's ID)
SELECT 
  client_name
FROM 
  `my-data-project-11097972.Dundler_Miffin.client` -- outer query
WHERE
  branch_id IN (
                 SELECT 
                   branch_id
                 FROM
                   `my-data-project-11097972.Dundler_Miffin.branch` --- second inner query
                  WHERE 
                    mgr_id IN (
                                 SELECT
                                   emp_id
                                 FROM
                                   `my-data-project-11097972.Dundler_Miffin.employee` -- build inner query
                                 WHERE 
                                   first_name = 'Michael' AND last_name = 'Scott'))

-- Find the names of all clients who have spent more than 100,000 dollars
SELECT 
  client_name
FROM
  `my-data-project-11097972.Dundler_Miffin.client`
WHERE
  client_id IN (
                SELECT client_id
                FROM (
                        SELECT 
                          SUM(total_sales) AS totals, 
                          client_id
                        FROM
                          `my-data-project-11097972.Dundler_Miffin.works_with`
                        GROUP BY 
                        client_id)
               WHERE totals > 100000)
